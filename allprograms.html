      }

      g2d.fillPolygon(starPolygon);
   }

public class ChineseFlag extends JPanel
{
   
   private static final int FLAG_WIDTH = 800;
   private static final int FLAG_HEIGHT = (int) (FLAG_WIDTH * 2.0 / 3.0); // Flag is 2/3 the width in height
   private static final int STAR_RADIUS = 15; // Radius of the circumcircle of the star
   private static final int OUTER_POINT_FACTOR = (int) (Math.cos(Math.PI / 10) / Math.cos(2 * Math.PI / 5)); // Factor to calculate outer points
   private static final int BIG_STAR_CENTER_X = FLAG_WIDTH / 4;
   private static final int BIG_STAR_CENTER_Y = FLAG_HEIGHT / 2;
   private static final int SMALL_STARS_OFFSET_X = BIG_STAR_CENTER_X - STAR_RADIUS * OUTER_POINT_FACTOR - 10;
   private static final int SMALL_STARS_OFFSET_Y = BIG_STAR_CENTER_Y - STAR_RADIUS - 10;
   
   @Override
   protected void paintComponent(Graphics g)
   {
      super.paintComponent(g);
      g.setColor(Color.RED);
      g.fillRect(0, 0, FLAG_WIDTH, FLAG_HEIGHT);
      drawStar(g, BIG_STAR_CENTER_X, BIG_STAR_CENTER_Y, STAR_RADIUS, Math.PI / 2, true);
      int[] smallStarX =
      {
         SMALL_STARS_OFFSET_X,
         SMALL_STARS_OFFSET_X + STAR_RADIUS * OUTER_POINT_FACTOR * 2 + 20,
         SMALL_STARS_OFFSET_X,
         SMALL_STARS_OFFSET_X + STAR_RADIUS * OUTER_POINT_FACTOR * 2 + 20
      };
      int[] smallStarY =
      {
         SMALL_STARS_OFFSET_Y,
         SMALL_STARS_OFFSET_Y,
         SMALL_STARS_OFFSET_Y + STAR_RADIUS * 2 + 20,
         SMALL_STARS_OFFSET_Y + STAR_RADIUS * 2 + 20
      };
      for (int i = 0; i < 4; i++)
      {
         drawStar(g, smallStarX[i], smallStarY[i], STAR_RADIUS, Math.PI / 4 + i * Math.PI / 2, false);
      }
   }
   private void drawStar(Graphics g, int centerX, int centerY, int radius, double rotation, boolean isBigStar)
   {
      Graphics2D g2d = (Graphics2D) g;
      g2d.translate(centerX, centerY);
      g2d.rotate(rotation);
      int outerRadius = radius * OUTER_POINT_FACTOR;
      int[] xPoints = new int[10];
      int[] yPoints = new int[10];
      for (int i = 0; i < 5; i++)
      {
         double angle = 2 * Math.PI * i / 5;
         xPoints[i * 2] = (int) (Math.cos(angle) * (isBigStar ? radius : outerRadius));
         yPoints[i * 2] = (int) (Math.sin(angle) * (isBigStar ? radius : outerRadius));
         
         xPoints[i * 2 + 1] = (int) (Math.cos(angle + Math.PI / 5) * outerRadius);
         yPoints[i * 2 + 1] = (int) (Math.sin(angle + Math.PI / 5) * outerRadius);
         if (!isBigStar)
         {
            xPoints[i * 2 + 1] += (xPoints[i * 2 + 1] - xPoints[i * 2]) / 5;
            yPoints[i * 2 + 1] += (yPoints[i * 2 + 1] - yPoints[i * 2]) / 5;
         }
      }
      if (isBigStar)
      {
         g2d.setColor(Color.YELLOW);
         g2d.fillPolygon(xPoints, yPoints, 5);
      }
      else
      {
         g2d.setColor(Color.YELLOW);
         g2d.fillPolygon(xPoints, yPoints, 10);
         g2d.setColor(Color.WHITE);
         g2d.fillPolygon(xPoints, yPoints, 5);
      }
      g2d.rotate(-rotation);
      g2d.translate(-centerX, -centerY);
   }
   public static void main(String[] args)
   {
      JFrame frame = new JFrame("Chinese Flag");
      ChineseFlag flagPanel = new ChineseFlag();
      frame.add(flagPanel);
      frame.setSize(FLAG_WIDTH, FLAG_HEIGHT);
      frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
      frame.setLocationRelativeTo(null); // Center the frame
      frame.setVisible(true);
   }
}

import javax.swing.*;
import java.awt.*;
import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;
import java.io.IOException;
import java.net.*;
public class UDPChatApp extends JFrame
{
   private JTextArea sendTextArea;
   private JTextArea receiveTextArea;
   private DatagramSocket socket;
   private InetAddress receiverAddress;
   private int port = 2311;
   public UDPChatApp()
   {
      setTitle("UDP Chat Application");
      setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
      setSize(600, 400);
      setLayout(new BorderLayout());
      sendTextArea = new JTextArea(5, 20);
      receiveTextArea = new JTextArea(15, 20);
      receiveTextArea.setEditable(false);
      JButton sendButton = new JButton("Send");
      sendButton.addActionListener(new ActionListener()
      {
         @Override
         public void actionPerformed(ActionEvent e)
         {
            sendMessage();
         }
      }
      );
      JPanel inputPanel = new JPanel();
      inputPanel.setLayout(new BorderLayout());
      inputPanel.add(new JScrollPane(sendTextArea), BorderLayout.CENTER);
      inputPanel.add(sendButton, BorderLayout.EAST);
      add(new JScrollPane(receiveTextArea), BorderLayout.CENTER);
      add(inputPanel, BorderLayout.SOUTH);
      setVisible(true);
      try
      {
         socket = new DatagramSocket(port);
         startListening();
      }
      catch (SocketException e)
      {
         e.printStackTrace();
      }
   }
   private void sendMessage()
   {
      String message = sendTextArea.getText();
      if (!message.isEmpty())
      {
         try
         {
            byte[] sendBuffer = message.getBytes();
            DatagramPacket sendPacket = new DatagramPacket(sendBuffer, sendBuffer.length, receiverAddress, port);
            socket.send(sendPacket);
            sendTextArea.setText("");
         }
         catch (IOException e)
         {
            e.printStackTrace();
         }
      }
   }
   private void startListening()
   {
      Thread receiveThread = new Thread(() ->
      {
         while (true)
         {
            try
            {
               byte[] receiveBuffer = new byte[1024];
               DatagramPacket receivePacket = new DatagramPacket(receiveBuffer, receiveBuffer.length);
               socket.receive(receivePacket);
               String receivedMessage = new String(receivePacket.getData(), 0, receivePacket.getLength());
               receiveTextArea.append(receivedMessage + "\n");
            }
            catch (IOException e)
            {
               e.printStackTrace();
            }
         }
      }
      );
      receiveThread.start();
   }
   public static void main(String[] args)
   {
      SwingUtilities.invokeLater(() ->
      {
         UDPChatApp app = new UDPChatApp();
         try
         {
            app.receiverAddress = InetAddress.getByName(JOptionPane.showInputDialog("Enter receiver's IP address"));
         }
         catch (UnknownHostException e)
         {
            e.printStackTrace();
         }
      }
      );
   }
}

         +"Content-Type:text/plain\r\n"
         +"\r\n"+
         "request received";
         out.writeBytes(response);
         in.close();
         out.close();
         clientSocket.close();
      }
      catch(IOException e)
      {
         e.printStackTrace();
      }
   }
   private Map readHeaders(BufferedReader in)throws IOException
   {
      Map headers = new HashMap<>();
      String line;
      while(!(line = in.readLine()).isEmpty())
      {
         String[]parts = line.split(":",2);
         if (parts.length == 2)
         {
            headers.put(parts[0].trim(), parts[1].trim());
         }
      }
      return headers;
   }

   {
      SimpleHTTPServer server = new SimpleHTTPServer(8080);
      server.start();
   }
